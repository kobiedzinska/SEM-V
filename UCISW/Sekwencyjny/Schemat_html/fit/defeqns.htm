********** Mapped Logic **********
FDCPE_Q0: FDCPE port map (Q(0),Q_D(0),CLK,CLR,'0',CE);
     Q_D(0) <= ((Q(1) AND NOT Q(0))
      OR (Q(1) AND NOT Q(2) AND NOT DIR)
      OR (NOT Q(1) AND Q(2) AND DIR)
      OR (NOT Q(0) AND NOT Q(2) AND NOT DIR));
FTCPE_Q1: FTCPE port map (Q(1),Q_T(1),CLK,CLR,'0',CE);
     Q_T(1) <= ((Q(0) AND NOT DIR)
      OR (NOT Q(1) AND Q(2) AND DIR)
      OR (Q(1) AND NOT Q(0) AND NOT Q(2) AND DIR));
FDCPE_Q2: FDCPE port map (Q(2),Q_D(2),CLK,CLR,'0',CE);
     Q_D(2) <= ((Q(1) AND Q(2))
      OR (Q(1) AND Q(0) AND NOT DIR)
      OR (NOT Q(1) AND NOT Q(0) AND DIR)
      OR (Q(0) AND Q(2) AND NOT DIR));
